程序入口参数 argc:参数数量 argv:参数数组
```
#include <iostream>
//程序入口参数 argc:参数数量 argv:参数数组
int main(int argc, char **argv)
{
    std::cout <<"参数数量=:"<< argc << std::endl;
    std::cout <<"程序名字=:"<< argv[0] << std::endl;
    std::string arg1 = argv[1];
    return 0;
}
```
